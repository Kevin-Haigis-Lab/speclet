def filter_hdi_across_zero(
    df: pd.DataFrame, hdi_low: str = "hdi_5.5%", hdi_high: str = "hdi_94.5%"
) -> pd.DataFrame:
    return df.query(
        f"(`{hdi_low}` < 0 and `{hdi_high}` < 0) or (`{hdi_low}` > 0 and `{hdi_high}` > 0)"
    ).reset_index(drop=True)
